Инициализация плагина 

<script>
$(function(){
	$('.demo5_text').liTranslit({
		translated:function(el, text, eventType){
			$('.info').remove();
			$('<span>').addClass('info').html(eventType+': '+text).insertAfter($('.demo5_text'))
		}
	});
});
</script>
Код HTML 

<body>
	<input type="text" class="demo5_text">
